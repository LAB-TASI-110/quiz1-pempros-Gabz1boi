void tampilkanMenu() {

long hitungHarga(int layanan, float berat) {
    long hargaPerKg;
    if (layanan == 1) hargaPerKg = 5000;      // Cuci Kering
    else if (layanan == 2) hargaPerKg = 7000; // Cuci Setrika
    else hargaPerKg = 10000;                  // Express
    
    return (long)(hargaPerKg * berat);
}

    Pesanan daftarPesanan[100];

            scanf("%d", &daftarPesanan[jumlahPesanan].jenisLayanan);

            daftarPesanan[jumlahPesanan].totalHarga = hitungHarga(daftarPesanan[jumlahPesanan].jenisLayanan, daftarPesanan[jumlahPesanan].berat);
            strcpy(daftarPesanan[jumlahPesanan].status, "Diproses");

#define HARGA_PER_BIJI 500 // Asumsi biaya per satuan barang (misal Rp 500)

typedef struct {
    char nama[50];
    char asrama[30];
    char tanggal[20];
    char listBarang[100];
    int jumlahBiji;
    long totalHarga;
} DataLaundry;

int main() {
    DataLaundry mhs;

    printf("=== Formulir Operasional Laundry Del ===\n");

    printf("Nama Mahasiswa    : ");
    fgets(mhs.nama, sizeof(mhs.nama), stdin);
    mhs.nama[strcspn(mhs.nama, "\n")] = 0;

    printf("Asrama            : ");
    fgets(mhs.asrama, sizeof(mhs.asrama), stdin);
    mhs.asrama[strcspn(mhs.asrama, "\n")] = 0;

    printf("Tanggal (DD/MM/YY): ");
    fgets(mhs.tanggal, sizeof(mhs.tanggal), stdin);
    mhs.tanggal[strcspn(mhs.tanggal, "\n")] = 0;

    printf("Daftar Barang     : ");
    fgets(mhs.listBarang, sizeof(mhs.listBarang), stdin);
    mhs.listBarang[strcspn(mhs.listBarang, "\n")] = 0;

    printf("Jumlah Barang (biji): ");
    scanf("%d", &mhs.jumlahBiji);

    // Perhitungan total berdasarkan jumlah biji
    mhs.totalHarga = (long)mhs.jumlahBiji * HARGA_PER_BIJI;

    printf("\n--- Ringkasan Pesanan Laundry ---\n");
    printf("Nama        : %s\n", mhs.nama);
    printf("Asrama      : %s\n", mhs.asrama);
    printf("Tanggal     : %s\n", mhs.tanggal);
    printf("Isi Paket   : %s\n", mhs.listBarang);
    printf("Total Item  : %d biji\n", mhs.jumlahBiji);
    printf("---------------------------------\n");
    printf("TOTAL TAGIHAN: Rp %ld\n", mhs.totalHarga);
    printf("---------------------------------\n");

    return 0;
}
